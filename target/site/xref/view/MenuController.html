31      /**
32       * Handles the start game button.
33       * @param event is an action event for the button
34       * @throws IOException if the input is not right.
35       */
36      @FXML
37      private void handleButtonAction(final ActionEvent event) throws IOException {
38          Node node = (Node) event.getSource();
39          Stage stage = (Stage) node.getScene().getWindow();
40          Parent root = FXMLLoader.load(getClass().getResource("/StartGame.fxml"));
41          Scene scene = new Scene(root);
42          stage.setScene(scene);
43          stage.show();
44      }
