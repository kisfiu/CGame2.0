42  
43  			Logger.info("Root element :" + doc.getDocumentElement().getNodeName());
44  
45  			NodeList nList = doc.getElementsByTagName("player");
46  
47  			Logger.info("----------------------------");
48  
49  			for (int temp = 0; temp < nList.getLength(); temp++)
50  			{
51  				Node nNode = nList.item(temp);
52  				Logger.info("\nCurrent Element :" + nNode.getNodeName());
53  				if (nNode.getNodeType() == Node.ELEMENT_NODE)
54  				{
55  					Element eElement = (Element) nNode;
56  					String neve = eElement.getAttribute("name");
57  					if (neve.equals(name1))
58  					{
59  						Logger.info("\n" + "It was already used." + "\n");
60  					}
61  
62  					String pontja = eElement.getElementsByTagName("points").item(0).getTextContent();
63  					Logger.info(pontja);
64  				}
65  			}
66      	}
67  		catch (Exception e)
68  		{
69  			e.printStackTrace();
70  			Logger.error("BAMM", e);
71      	}
