3   import java.io.IOException;
4   
5   import javafx.application.Application;
6   import javafx.fxml.FXMLLoader;
7   import javafx.scene.Scene;
8   import javafx.scene.layout.AnchorPane;
9   import javafx.stage.Stage;
10  import org.pmw.tinylog.Logger;
11  
12  /**
13   * Main app.
14   * Extends application.Application
15   * @author kisfiu
16   */
17  public class MainApp extends Application {
18      /**
19       * Stage Variable primaryStage declares the primary stage that we are using.
20       * AnchorPane Variable menu is the main menu
21       */
22      private Stage primaryStage; private AnchorPane menu;
23  
24       /**
25       * @param primaryStage is the UI's primary stage
26       */
27      @Override
28      public final void start(final Stage primaryStage) {
29          this.primaryStage = primaryStage;
30          this.primaryStage.setTitle("ColorGame");
31          showmenu();
32      }
33  
34  
35  
36      /**
37       * Initializes the root layout.
38       */
39      public final void showmenu() {
40          try {
41              // Load root layout from fxml file.
42              FXMLLoader loader = new FXMLLoader();
43              loader.setLocation(MainApp.class.getResource("/Menu.fxml"));
44              menu = (AnchorPane) loader.load();
45  
46              Logger.info("My first log entry");
47              Logger.warn("Test warning.");
48  
49              // Show the scene containing the root layout.
50              Scene scene = new Scene(menu);
51              primaryStage.setScene(scene);
52              primaryStage.show();
53          } catch (IOException e) {
54              e.printStackTrace();
55              Logger.error("BAMM Exception :: ", e);
56          }
57      }
58  
59  
60      /**
61       * Returns the main stage.
62       * @return primaryStage
63       */
64      public final Stage getPrimaryStage() {
65          return primaryStage;
66      }
67  
68      /**
69       * @param args Any arguments to be included in a string representation of this event.
70       */
71      public static void main(final String[] args) {
72          launch(args);
73      }
74  }
